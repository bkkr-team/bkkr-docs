Usage

Angular

import { Component } from '@angular/core';
import { LoadingController } from '@bkkr/angular';

@Component({
  selector: 'loading-example',
  templateUrl: 'loading-example.html',
  styleUrls: ['./loading-example.css']
})
export class LoadingExample {
  constructor(public loadingController: LoadingController) {

  }

  async presentLoading() {
    const loading = await this.loadingController.create({
      cssClass: 'my-custom-class'
    });
    return await loading.present();
  }
}

Javascript

function presentLoading() {
  const loadingElement = document.createElement('bkkr-loading');
  loadingElement.cssClass = 'my-custom-class';

  // present the loading
  document.body.appendChild(loadingElement);
  return loadingElement.present();
}

Property Attribute Description Type Default
animated animated If true, the loading will animate. boolean true
color color The color to use from your application’s color palette. Default options are: "primary", "secondary", "tertiary", "success", "warning", "danger", "light", "medium", and "dark". For more information on colors, see theming. string | undefined 'primary'
cssClass css-class Additional classes to apply for custom CSS. If multiple classes are provided they should be separated by spaces. string | string[] | undefined undefined
enterAnimation – Animation to use when the loading is presented. ((baseEl: any, opts?: any) => Animation) | undefined undefined
htmlAttributes – Additional attributes to pass to the loader. LoadingAttributes | undefined undefined
keyboardClose keyboard-close If true, the keyboard will be automatically dismissed when the overlay is presented. boolean false
leaveAnimation – Animation to use when the loading is dismissed. ((baseEl: any, opts?: any) => Animation) | undefined undefined
position position The position of the loading on the screen. "bottom" | "middle" | "top" 'top'
spinner spinner The type of spinner being used in loader. "circles" | "circular" | "crescent" | "dots" | "lines" | "lines-sharp" | "lines-sharp-small" | "lines-small" | undefined undefined
