<!-- Default -->
<bkkr-card>
  <bkkr-card-header>
    <bkkr-card-subtitle>Card Subtitle</bkkr-card-subtitle>
    <h2 class="mb-0 text-bold">Card Title</h2>

    <bkkr-button fill="outline" slot="end">View</bkkr-button>
  </bkkr-item>

  <bkkr-card-content>
    This is content, without any paragraph or header tags,
    within an bkkr-card-content element.
  </bkkr-card-content>
</bkkr-card>
