bkkr-modal

import { Component } from '@angular/core';
import { ModalController } from '@bkkr/angular';
import { ModalPage } from '../modal/modal.page';

@Component({
  selector: 'modal-example',
  templateUrl: 'modal-example.html',
  styleUrls: ['./modal-example.css']
})
export class ModalExample {
  constructor(public modalController: ModalController) {

  }

  async presentModal() {
    const modal = await this.modalController.create({
      component: ModalPage,
      cssClass: 'my-custom-class'
    });
    return await modal.present();
  }
}
customElements.define('modal-page', class extends HTMLElement {
  connectedCallback() {
    this.innerHTML = `
<bkkr-header>
  <bkkr-toolbar>
    <bkkr-title>Modal Header</bkkr-title>
    <bkkr-buttons slot="primary">
      <bkkr-button onClick="dismissModal()">
        <bkkr-icon slot="icon-only" name="close"></bkkr-icon>
      </bkkr-button>
    </bkkr-buttons>
  </bkkr-toolbar>
</bkkr-header>
<bkkr-content class="bkkr-padding">
  Modal Content
</bkkr-content>`;
  }
});

function presentModal() {
  // create the modal with the `modal-page` component
  const modalElement = document.createElement('bkkr-modal');
  modalElement.component = 'modal-page';
  modalElement.cssClass = 'my-custom-class';

  // present the modal
  document.body.appendChild(modalElement);
  return modalElement.present();
}
