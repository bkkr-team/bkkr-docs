import { Component } from '@angular/core';
import { PopoverController } from '@bkkr/angular';
import { PopoverPage } from '../popover/popover.page';

@Component({
  selector: 'popover-example',
  templateUrl: 'popover-example.html',
  styleUrls: ['./popover-example.css']
})
export class PopoverExample {
  constructor(public popoverController: PopoverController) {

  }

  async presentPopover() {
    const popover = await this.popoverController.create({
      component: PopoverPage,
      event: ev,
      cssClass: 'my-custom-class'
    });
    return await popover.present();
  }
}
customElements.define('popover-page', class extends HTMLElement {
  connectedCallback() {
    this.innerHTML = `
<bkkr-header>
  <bkkr-toolbar>
    <bkkr-title>Popover Header</bkkr-title>
    <bkkr-buttons slot="primary">
      <bkkr-button onClick="dismissPopover()">
        <bkkr-icon slot="icon-only" name="close"></bkkr-icon>
      </bkkr-button>
    </bkkr-buttons>
  </bkkr-toolbar>
</bkkr-header>
<bkkr-content class="bkkr-padding">
  Popover Content
</bkkr-content>`;
  }
});

function presentPopover() {
  // create the popover with the `popover-page` component
  const popoverElement = document.createElement('bkkr-popover');
  popoverElement.component = 'popover-page';
  popoverElement.cssClass = 'my-custom-class';

  // present the popover
  document.body.appendChild(popoverElement);
  return popoverElement.present();
}

triggerActionâ€‹

DescriptionDescribes what kind of interaction with the trigger that
should cause the popover to open. Does not apply when the trigger
property is undefined.
If 'click', the popover will be presented when the trigger is left clicked.
If 'hover', the popover will be presented when a pointer hovers over the trigger.
If 'context-menu', the popover will be presented when the trigger is right
clicked on desktop and long pressed on mobile. This will also prevent your
device's normal context menu from appearing.
Attributetrigger-action
Type"click" ï½œ "context-menu" ï½œ "hover"
