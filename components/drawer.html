import { Component } from '@angular/core';
import { DrawerController } from '@bkkr/angular';
import { DrawerPage } from '../drawer/drawer.page';

@Component({
  selector: 'drawer-example',
  templateUrl: 'drawer-example.html',
  styleUrls: ['./drawer-example.css']
})
export class DrawerExample {
  constructor(public drawerController: DrawerController) {

  }

  async presentDrawer() {
    const drawer = await this.drawerController.create({
      component: DrawerPage,
      cssClass: 'my-custom-class'
    });
    return await drawer.present();
  }
}
customElements.define('drawer-page', class extends HTMLElement {
  connectedCallback() {
    this.innerHTML = `
<bkkr-header>
  <bkkr-toolbar>
    <bkkr-title>Drawer Header</bkkr-title>
    <bkkr-buttons slot="primary">
      <bkkr-button onClick="dismissDrawer()">
        <bkkr-icon slot="icon-only" name="close"></bkkr-icon>
      </bkkr-button>
    </bkkr-buttons>
  </bkkr-toolbar>
</bkkr-header>
<bkkr-content class="bkkr-padding">
  Drawer Content
</bkkr-content>`;
  }
});

function presentDrawer() {
  // create the drawer with the `drawer-page` component
  const drawerElement = document.createElement('bkkr-drawer');
  drawerElement.component = 'drawer-page';
  drawerElement.cssClass = 'my-custom-class';

  // present the drawer
  document.body.appendChild(drawerElement);
  return drawerElement.present();
}
