  <!--  the side menu  -->
  <bkkr-menu content-id="main">
    <bkkr-header>
      <bkkr-toolbar>
        <bkkr-title>Menu</bkkr-title>
      </bkkr-toolbar>
    </bkkr-header>
  </bkkr-menu>

  <!-- the main content -->
  <bkkr-content id="main">
    <h1>Hello</h1>
  </bkkr-content>
</bkkr-split-pane>
