bkkr-content

Usageâ€‹

  • Angular
  • Javascript
<!-- Default -->
<bkkr-content
  [scrollEvents]="true"
  (bkkrScrollStart)="logScrollStart()"
  (bkkrScroll)="logScrolling($event)"
  (bkkrScrollEnd)="logScrollEnd()">
    <h1>Main Content</h1>

    <div slot="fixed">
      <h1>Fixed Content</h1>
    </div>
</bkkr-content>
<!-- Default -->
<bkkr-content>
  <h1>Main Content</h1>

  <div slot="fixed">
    <h1>Fixed Content</h1>
  </div>
</bkkr-content>

<script>
  var content = document.querySelector('bkkr-content');
  content.scrollEvents = true;

  content.addEventListener('bkkrScrollStart', () => console.log('scroll start'));

  content.addEventListener('bkkrScroll', (ev) => console.log('scroll', ev.detail));

  content.addEventListener('bkkrScrollEnd', () => console.log('scroll end'));
</script>

DescriptionGet the element where the actual scrolling takes place.
This element can be used to subscribe to scroll events or manually modify
scrollTop. However, it's recommended to use the API provided by bkkr-content:
